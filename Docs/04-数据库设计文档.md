## 1. 概述

### 1.1 设计目标
本文档详细描述了AI新媒体专家系统的数据库设计，包括数据模型、表结构、索引策略、约束关系等。数据库设计遵循第三范式，确保数据的一致性、完整性和高效性。

### 1.2 设计原则
- **数据完整性**：通过约束和触发器确保数据的准确性
- **性能优化**：合理的索引设计和查询优化
- **可扩展性**：支持业务增长和功能扩展
- **安全性**：敏感数据加密和访问控制
- **可维护性**：清晰的命名规范和文档说明

### 1.3 技术选型
- **主数据库**: PostgreSQL 13+
- **缓存数据库**: Redis 6+
- **搜索引擎**: Elasticsearch 7+
- **时序数据库**: InfluxDB 2.0+
- **对象存储**: MinIO (S3兼容)

#### 2.2.1 核心业务数据 (PostgreSQL)

CREATE OR REPLACE FUNCTION update_updated_at_column()

BEGIN
    NEW.updated_at = CURRENT_TIMESTAMP;
    RETURN NEW;
END;
$$ language 'plpgsql';

    FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();

-- 创建触发器更新标签使用次数
CREATE OR REPLACE FUNCTION update_tag_usage_count()
RETURNS TRIGGER AS $$
BEGIN
    IF TG_OP = 'INSERT' THEN
        UPDATE tags SET usage_count = usage_count + 1 WHERE id = NEW.tag_id;
        RETURN NEW;
    ELSIF TG_OP = 'DELETE' THEN
        UPDATE tags SET usage_count = usage_count - 1 WHERE id = OLD.tag_id;
        RETURN OLD;
    END IF;
    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trigger_update_tag_usage_count
    AFTER INSERT OR DELETE ON video_tags
    FOR EACH ROW EXECUTE FUNCTION update_tag_usage_count();
```

import redis
import json
from typing import Any, Optional

class CacheManager:
    def __init__(self, redis_url: str):
        self.redis = redis.from_url(redis_url)
        self.config = CACHE_CONFIG
    
    def get(self, cache_type: str, key: str) -> Optional[Any]:
        """获取缓存数据"""
        config = self.config.get(cache_type, {})
        full_key = f"{config.get('prefix', '')}{key}"
        
        data = self.redis.get(full_key)
        if data:
            return json.loads(data)
        return None
    
    def set(self, cache_type: str, key: str, value: Any, ttl: Optional[int] = None) -> bool:
        """设置缓存数据"""
        config = self.config.get(cache_type, {})
        full_key = f"{config.get('prefix', '')}{key}"
        ttl = ttl or config.get('ttl', 3600)
        
        return self.redis.setex(full_key, ttl, json.dumps(value))
    
    def delete(self, cache_type: str, key: str) -> bool:
        """删除缓存数据"""
        config = self.config.get(cache_type, {})
        full_key = f"{config.get('prefix', '')}{key}"
        
        return bool(self.redis.delete(full_key))
    
    def exists(self, cache_type: str, key: str) -> bool:
        """检查缓存是否存在"""
        config = self.config.get(cache_type, {})
        full_key = f"{config.get('prefix', '')}{key}"
        
        return bool(self.redis.exists(full_key))

from elasticsearch import Elasticsearch
from typing import Dict, List, Any

class SearchService:
    def __init__(self, es_url: str):
        self.es = Elasticsearch([es_url])
    
    def search_videos(self, query: str, filters: Dict = None, page: int = 1, size: int = 20) -> Dict:
        """搜索视频"""
        body = {
            "query": {
                "bool": {
                    "must": [
                        {
                            "multi_match": {
                                "query": query,
                                "fields": ["title^3", "description^2", "author", "tags"]
                            }
                        }
                    ],
                    "filter": []
                }
            },
            "sort": [{"created_at": {"order": "desc"}}],
            "from": (page - 1) * size,
            "size": size,
            "highlight": {
                "fields": {
                    "title": {},
                    "description": {}
                }
            }
        }
        
        # 添加过滤条件
        if filters:
            if filters.get('platform'):
                body['query']['bool']['filter'].append({
                    "term": {"platform": filters['platform']}
                })
            
            if filters.get('date_range'):
                body['query']['bool']['filter'].append({
                    "range": {
                        "upload_date": filters['date_range']
                    }
                })
            
            if filters.get('tags'):
                body['query']['bool']['filter'].append({
                    "terms": {"tags": filters['tags']}
                })
        
        return self.es.search(index="videos", body=body)
    
    def suggest_tags(self, query: str, size: int = 10) -> List[str]:
        """标签建议"""
        body = {
            "suggest": {
                "tag_suggest": {
                    "prefix": query,
                    "completion": {
                        "field": "tags.suggest",
                        "size": size
                    }
                }
            }
        }
        
        result = self.es.search(index="videos", body=body)
        suggestions = []
        
        for suggestion in result['suggest']['tag_suggest'][0]['options']:
            suggestions.append(suggestion['text'])
        
        return suggestions
```

    start_date = f"{year}-{month:02d}-01"

        end_date = f"{year + 1}-01-01"
