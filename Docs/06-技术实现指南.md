```python
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker, Session
from sqlalchemy.pool import StaticPool
from .config import settings
import logging

logger = logging.getLogger(__name__)

# 根据数据库类型创建引擎
def create_database_engine():
    """根据配置自动选择数据库引擎"""
    if settings.DATABASE_URL.startswith('sqlite'):
        # SQLite配置
        engine = create_engine(
            settings.DATABASE_URL,
            poolclass=StaticPool,
            connect_args={
                "check_same_thread": False,  # SQLite特殊配置
                "timeout": 20
            },
            echo=settings.DEBUG
        )
        logger.info("Using SQLite database for development")
    else:
        # PostgreSQL配置
        engine = create_engine(
            settings.DATABASE_URL,
            pool_pre_ping=True,
            pool_recycle=300,
            pool_size=10,
            max_overflow=20,
            echo=settings.DEBUG
        )
        logger.info("Using PostgreSQL database for production")
    
    return engine

# 创建数据库引擎
engine = create_database_engine()

# 创建会话工厂
SessionLocal = sessionmaker(autocommit=False, autoflush=False, bind=engine)

def get_db() -> Session:
    """获取数据库会话"""
    db = SessionLocal()
    try:
        yield db
    except Exception as e:
        logger.error(f"Database error: {e}")
        db.rollback()
        raise
    finally:
        db.close()

def init_db():
    """初始化数据库"""
    from app.models.base import Base
    Base.metadata.create_all(bind=engine)
    logger.info("Database initialized")
```

#### 3.3.1 密码处理

```python
# app/core/security.py
from passlib.context import CryptContext
from jose import JWTError, jwt
from datetime import datetime, timedelta
from .config import settings
import logging

logger = logging.getLogger(__name__)

# 密码加密上下文
pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")

def verify_password(plain_password: str, hashed_password: str) -> bool:
    """验证密码"""
    return pwd_context.verify(plain_password, hashed_password)

def get_password_hash(password: str) -> str:
    """生成密码哈希"""
    return pwd_context.hash(password)

def create_access_token(data: dict, expires_delta: timedelta = None) -> str:
    """创建访问令牌"""
    to_encode = data.copy()
    if expires_delta:
        expire = datetime.utcnow() + expires_delta
    else:
        expire = datetime.utcnow() + timedelta(minutes=settings.ACCESS_TOKEN_EXPIRE_MINUTES)
    
    to_encode.update({"exp": expire, "type": "access"})
    encoded_jwt = jwt.encode(to_encode, settings.SECRET_KEY, algorithm=settings.ALGORITHM)
    return encoded_jwt

def create_refresh_token(data: dict) -> str:
    """创建刷新令牌"""
    to_encode = data.copy()
    expire = datetime.utcnow() + timedelta(days=settings.REFRESH_TOKEN_EXPIRE_DAYS)
    to_encode.update({"exp": expire, "type": "refresh"})
    encoded_jwt = jwt.encode(to_encode, settings.SECRET_KEY, algorithm=settings.ALGORITHM)
    return encoded_jwt

def verify_token(token: str) -> dict:
    """验证令牌"""
    try:
        payload = jwt.decode(token, settings.SECRET_KEY, algorithms=[settings.ALGORITHM])
        return payload
    except JWTError as e:
        logger.error(f"Token verification error: {e}")
        return None
```

#### 3.3.2 认证依赖

```python
# app/core/deps.py
from fastapi import Depends, HTTPException, status
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials
from sqlalchemy.orm import Session
from .database import get_db
from .security import verify_token
from app.models.user import User
import logging

logger = logging.getLogger(__name__)
security = HTTPBearer()

def get_current_user(
    credentials: HTTPAuthorizationCredentials = Depends(security),
    db: Session = Depends(get_db)
) -> User:
    """获取当前用户"""
    token = credentials.credentials
    payload = verify_token(token)
    
    if payload is None:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Invalid token",
            headers={"WWW-Authenticate": "Bearer"},
        )
    
    user_id = payload.get("sub")
    if user_id is None:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Invalid token",
            headers={"WWW-Authenticate": "Bearer"},
        )
    
    user = db.query(User).filter(User.id == user_id).first()
    if user is None:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="User not found",
            headers={"WWW-Authenticate": "Bearer"},
        )
    
    return user

def get_current_active_user(current_user: User = Depends(get_current_user)) -> User:
    """获取当前活跃用户"""
    if current_user.status != "active":
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail="Inactive user"
        )
    return current_user
```

```python
# app/services/video_service.py
from sqlalchemy.orm import Session
from sqlalchemy import and_, or_
from fastapi import UploadFile, HTTPException, status
from app.models.video import Video
from app.core.config import settings
from typing import List, Tuple, Dict, Optional
import os
import hashlib
import aiofiles
import cv2
from PIL import Image
import logging

logger = logging.getLogger(__name__)

class VideoService:
    def __init__(self, db: Session):
        self.db = db
    
    def get_videos(
        self,
        page: int = 1,
        size: int = 20,
        filters: Dict = None,
        user_id: int = None
    ) -> Tuple[List[Video], int]:
        """获取视频列表"""
        query = self.db.query(Video).filter(Video.status == "active")
        
        # 应用过滤条件
        if filters:
            if filters.get('platform'):
                query = query.filter(Video.platform == filters['platform'])
            
            if filters.get('author'):
                query = query.filter(Video.author.ilike(f"%{filters['author']}%"))
            
            if filters.get('search'):
                search_term = f"%{filters['search']}%"
                query = query.filter(
                    or_(
                        Video.title.ilike(search_term),
                        Video.description.ilike(search_term),
                        Video.author.ilike(search_term)
                    )
                )
        
        # 获取总数
        total = query.count()
        
        # 分页
        offset = (page - 1) * size
        videos = query.order_by(Video.created_at.desc()).offset(offset).limit(size).all()
        
        return videos, total
    
    async def upload_video(
        self,
        file: UploadFile,
        title: str,
        description: str = None,
        tags: List[str] = None,
        category: str = None,
        user_id: int = None
    ) -> Video:
        """上传视频文件"""
        # 验证文件类型
        if not file.content_type.startswith('video/'):
            raise HTTPException(
                status_code=status.HTTP_400_BAD_REQUEST,
                detail="Invalid file type. Only video files are allowed."
            )
        
        # 验证文件大小
        file_size = 0
        content = await file.read()
        file_size = len(content)
        
        if file_size > settings.MAX_FILE_SIZE:
            raise HTTPException(
                status_code=status.HTTP_400_BAD_REQUEST,
                detail=f"File size exceeds maximum limit of {settings.MAX_FILE_SIZE} bytes"
            )
        
        # 生成文件哈希
        file_hash = hashlib.sha256(content).hexdigest()
        
        # 检查文件是否已存在
        existing_video = self.db.query(Video).filter(Video.file_hash == file_hash).first()
        if existing_video:
            raise HTTPException(
                status_code=status.HTTP_409_CONFLICT,
                detail="Video already exists"
            )
        
        # 生成文件路径
        file_extension = os.path.splitext(file.filename)[1]
        file_name = f"{file_hash}{file_extension}"
        file_path = os.path.join(settings.UPLOAD_DIR, "videos", file_name)
        
        # 确保目录存在
        os.makedirs(os.path.dirname(file_path), exist_ok=True)
        
        # 保存文件
        async with aiofiles.open(file_path, 'wb') as f:
            await f.write(content)
        
        # 提取视频信息
        video_info = self._extract_video_info(file_path)
        
        # 生成缩略图
        thumbnail_path = await self._generate_thumbnail(file_path, file_hash)
        
        # 创建视频记录
        video = Video(
            title=title,
            description=description,
            file_path=file_path,
            file_name=file.filename,
            file_size=file_size,
            file_hash=file_hash,
            duration=video_info.get('duration'),
            width=video_info.get('width'),
            height=video_info.get('height'),
            resolution=f"{video_info.get('width')}x{video_info.get('height')}",
            fps=video_info.get('fps'),
            format=video_info.get('format'),
            codec_video=video_info.get('codec_video'),
            codec_audio=video_info.get('codec_audio'),
            thumbnail_path=thumbnail_path,
            tags=tags or [],
            category=category,
            created_by=user_id,
            status="active"
        )
        
        self.db.add(video)
        self.db.commit()
        self.db.refresh(video)
        
        logger.info(f"Video uploaded successfully: {video.id}")
        
        return video
    
    def _extract_video_info(self, file_path: str) -> Dict:
        """提取视频信息"""
        try:
            cap = cv2.VideoCapture(file_path)
            
            if not cap.isOpened():
                logger.error(f"Cannot open video file: {file_path}")
                return {}
            
            # 获取视频属性
            fps = cap.get(cv2.CAP_PROP_FPS)
            frame_count = int(cap.get(cv2.CAP_PROP_FRAME_COUNT))
            width = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))
            height = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))
            duration = frame_count / fps if fps > 0 else 0
            
            cap.release()
            
            return {
                'duration': int(duration),
                'width': width,
                'height': height,
                'fps': round(fps, 2),
                'frame_count': frame_count,
                'format': os.path.splitext(file_path)[1][1:].lower()
            }
            
        except Exception as e:
            logger.error(f"Error extracting video info: {e}")
            return {}
    
    async def _generate_thumbnail(self, video_path: str, file_hash: str) -> str:
        """生成视频缩略图"""
        try:
            cap = cv2.VideoCapture(video_path)
            
            if not cap.isOpened():
                logger.error(f"Cannot open video for thumbnail: {video_path}")
                return None
            
            # 跳到视频中间位置
            frame_count = int(cap.get(cv2.CAP_PROP_FRAME_COUNT))
            cap.set(cv2.CAP_PROP_POS_FRAMES, frame_count // 2)
            
            ret, frame = cap.read()
            cap.release()
            
            if not ret:
                logger.error(f"Cannot read frame for thumbnail: {video_path}")
                return None
            
            # 调整图片大小
            height, width = frame.shape[:2]
            if width > 320:
                ratio = 320 / width
                new_width = 320
                new_height = int(height * ratio)
                frame = cv2.resize(frame, (new_width, new_height))
            
            # 保存缩略图
            thumbnail_name = f"{file_hash}_thumb.jpg"
            thumbnail_path = os.path.join(settings.UPLOAD_DIR, "thumbnails", thumbnail_name)
            
            os.makedirs(os.path.dirname(thumbnail_path), exist_ok=True)
            cv2.imwrite(thumbnail_path, frame)
            
            logger.info(f"Thumbnail generated: {thumbnail_path}")
            
            return thumbnail_path
            
        except Exception as e:
            logger.error(f"Error generating thumbnail: {e}")
            return None
```

```python
from sqlalchemy.orm import Session
from app.models.download_task import DownloadTask
from app.models.video import Video

from typing import Dict, Optional
import yt_dlp
import os
import logging

logger = logging.getLogger(__name__)

class DownloadService:
    def __init__(self, db: Session):
        self.db = db
    
    def create_download_task(
        self,
        url: str,
        user_id: int,
        options: Dict = None,
        priority: int = 5
    ) -> DownloadTask:
        """创建下载任务"""
        # 检查URL是否已存在
        existing_task = self.db.query(DownloadTask).filter(
            DownloadTask.url == url,
            DownloadTask.status.in_(["pending", "processing"])
        ).first()
        
        if existing_task:
            logger.warning(f"Download task already exists for URL: {url}")
            return existing_task
        
        # 提取平台信息
        platform = self._detect_platform(url)
        
        # 创建任务
        task = DownloadTask(
            user_id=user_id,
            url=url,
            platform=platform,
            options=options or {},
            priority=priority,
            status="pending"
        )
        
        self.db.add(task)
        self.db.commit()
        self.db.refresh(task)
        
        logger.info(f"Download task created: {task.id}")
        
        return task
    
    def _detect_platform(self, url: str) -> str:
        """检测视频平台"""
        if "tiktok.com" in url or "douyin.com" in url:
            return "tiktok"
        elif "youtube.com" in url or "youtu.be" in url:
            return "youtube"
        elif "bilibili.com" in url:
            return "bilibili"
        elif "instagram.com" in url:
            return "instagram"
        else:
            return "unknown"
    
    def extract_video_info(self, url: str) -> Dict:
        """提取视频信息"""
        ydl_opts = {
            'quiet': True,
            'no_warnings': True,
            'extract_flat': False,
        }
        
        try:
            with yt_dlp.YoutubeDL(ydl_opts) as ydl:
                info = ydl.extract_info(url, download=False)
                
                return {
                    'title': info.get('title', ''),
                    'description': info.get('description', ''),
                    'duration': info.get('duration', 0),
                    'uploader': info.get('uploader', ''),
                    'upload_date': info.get('upload_date', ''),
                    'view_count': info.get('view_count', 0),
                    'like_count': info.get('like_count', 0),
                    'thumbnail': info.get('thumbnail', ''),
                    'formats': info.get('formats', []),
                    'platform_id': info.get('id', ''),
                    'webpage_url': info.get('webpage_url', url)
                }
                
        except Exception as e:
            logger.error(f"Error extracting video info: {e}")
            return {}
    
    def download_video(
        self,
        task_id: int,
        progress_callback=None
    ) -> Optional[Video]:
        """下载视频"""
        task = self.db.query(DownloadTask).filter(DownloadTask.id == task_id).first()
        
        if not task:
            logger.error(f"Download task not found: {task_id}")
            return None
        
        try:
            # 更新任务状态
            task.status = "processing"
            task.progress = 0
            self.db.commit()
            
            # 提取视频信息
            video_info = self.extract_video_info(task.url)
            
            if not video_info:
                task.status = "failed"
                task.error_message = "Failed to extract video information"
                self.db.commit()
                return None
            
            # 设置下载选项
            output_dir = os.path.join(settings.UPLOAD_DIR, "downloads")
            os.makedirs(output_dir, exist_ok=True)
            
            ydl_opts = {
                'outtmpl': os.path.join(output_dir, '%(title)s.%(ext)s'),
                'format': task.options.get('quality', 'best'),
                'writesubtitles': task.options.get('download_subtitles', False),
                'writeautomaticsub': task.options.get('download_subtitles', False),
                'progress_hooks': [lambda d: self._progress_hook(d, task_id, progress_callback)],
            }
            
            # 下载视频
            with yt_dlp.YoutubeDL(ydl_opts) as ydl:
                ydl.download([task.url])
            
            # 创建视频记录
            video = self._create_video_record(task, video_info, output_dir)
            
            # 更新任务状态
            task.status = "completed"
            task.progress = 100
            task.video_id = video.id if video else None
            self.db.commit()
            
            logger.info(f"Video downloaded successfully: {task_id}")
            
            return video
            
        except Exception as e:
            logger.error(f"Error downloading video: {e}")
            
            task.status = "failed"
            task.error_message = str(e)
            task.retry_count += 1
            self.db.commit()
            
            return None
    
    def _progress_hook(self, d, task_id: int, callback=None):
        """下载进度回调"""
        if d['status'] == 'downloading':
            if 'total_bytes' in d and d['total_bytes']:
                progress = int((d['downloaded_bytes'] / d['total_bytes']) * 100)
            elif '_percent_str' in d:
                progress = int(float(d['_percent_str'].replace('%', '')))
            else:
                progress = 0
            
            # 更新数据库中的进度
            task = self.db.query(DownloadTask).filter(DownloadTask.id == task_id).first()
            if task:
                task.progress = progress
                task.downloaded_size = d.get('downloaded_bytes', 0)
                task.file_size = d.get('total_bytes', 0)
                task.download_speed = d.get('speed', 0)
                task.eta = d.get('eta', 0)
                self.db.commit()
            
            # 调用外部回调
            if callback:
                callback({
                    'task_id': task_id,
                    'progress': progress,
                    'downloaded_bytes': d.get('downloaded_bytes', 0),
                    'total_bytes': d.get('total_bytes', 0),
                    'speed': d.get('speed', 0),
                    'eta': d.get('eta', 0)
                })
    
    def _create_video_record(self, task: DownloadTask, video_info: Dict, output_dir: str) -> Optional[Video]:
        """创建视频记录"""
        try:
            # 查找下载的文件
            title = video_info.get('title', 'Unknown')
            possible_files = []
            
            for ext in ['.mp4', '.mkv', '.webm', '.avi']:
                file_path = os.path.join(output_dir, f"{title}{ext}")
                if os.path.exists(file_path):
                    possible_files.append(file_path)
            
            if not possible_files:
                logger.error(f"Downloaded file not found for task: {task.id}")
                return None
            
            file_path = possible_files[0]
            file_size = os.path.getsize(file_path)
            
            # 创建视频记录
            video = Video(
                title=video_info.get('title', ''),
                description=video_info.get('description', ''),
                file_path=file_path,
                file_name=os.path.basename(file_path),
                file_size=file_size,
                duration=video_info.get('duration', 0),
                platform=task.platform,
                platform_id=video_info.get('platform_id', ''),
                original_url=task.url,
                author=video_info.get('uploader', ''),
                view_count=video_info.get('view_count', 0),
                like_count=video_info.get('like_count', 0),
                created_by=task.user_id,

            )
            
            self.db.add(video)
            self.db.commit()
            self.db.refresh(video)

        except Exception as e:
```

```python
    try:
        download_service = DownloadService(db)
        
        def progress_callback(progress_data):
            # 更新Celery任务状态
            current_task.update_state(
                state='PROGRESS',
                meta=progress_data
            )
        
        video = download_service.download_video(
            task_id=task_id,
            progress_callback=progress_callback
        )
        
        if video:
            return {
                'status': 'completed',
                'video_id': video.id,
                'message': 'Video downloaded successfully'
            }
        else:
            return {
                'status': 'failed',
                'message': 'Failed to download video'
            }
            
    except Exception as e:
        logger.error(f"Download task error: {e}")
        return {
            'status': 'failed',
            'message': str(e)
        }
    
    finally:
        db.close()

@celery_app.task
def batch_download_task(urls: list, user_id: int, options: dict = None):
    """批量下载任务"""
    db = SessionLocal()
    
    try:
        download_service = DownloadService(db)
        results = []
        
        for url in urls:
            try:
                task = download_service.create_download_task(
                    url=url,
                    user_id=user_id,
                    options=options
                )
                
                # 启动下载任务
                download_video_task.delay(task.id)
                
                results.append({
                    'url': url,
                    'task_id': task.id,
                    'status': 'queued'
                })
                
            except Exception as e:
                logger.error(f"Error creating download task for {url}: {e}")
                results.append({
                    'url': url,
                    'status': 'failed',
                    'error': str(e)
                })
        
        return {
            'status': 'completed',
            'results': results,
            'total': len(urls),
            'successful': len([r for r in results if r['status'] == 'queued'])
        }
        
    except Exception as e:
        logger.error(f"Batch download task error: {e}")
        return {
            'status': 'failed',
            'message': str(e)
        }
    
    finally:
        db.close()
```

```javascript
    this.listeners = new Map();
  }

  connect() {
    const token = localStorage.getItem('access_token');
    
    this.socket = io(process.env.REACT_APP_WS_URL || 'ws://localhost:8000', {
      auth: {
        token: token
      },
      transports: ['websocket']
    });

    this.socket.on('connect', () => {
      console.log('WebSocket connected');
    });

    this.socket.on('disconnect', () => {
      console.log('WebSocket disconnected');
    });

    this.socket.on('task_progress', (data) => {
      this.emit('task_progress', data);
    });

    this.socket.on('task_completed', (data) => {
      message.success(`任务 ${data.task_id} 已完成`);
      this.emit('task_completed', data);
    });

    this.socket.on('task_failed', (data) => {
      message.error(`任务 ${data.task_id} 失败: ${data.error}`);
      this.emit('task_failed', data);
    });
  }

  disconnect() {
    if (this.socket) {
      this.socket.disconnect();
      this.socket = null;
    }
  }

  on(event, callback) {
    if (!this.listeners.has(event)) {
      this.listeners.set(event, []);
    }
    this.listeners.get(event).push(callback);
  }

  off(event, callback) {
    if (this.listeners.has(event)) {
      const callbacks = this.listeners.get(event);
      const index = callbacks.indexOf(callback);
      if (index > -1) {
        callbacks.splice(index, 1);
      }
    }
  }

  emit(event, data) {
    if (this.listeners.has(event)) {
      this.listeners.get(event).forEach(callback => {
        callback(data);
      });
    }
  }

  joinRoom(room) {
    if (this.socket) {
      this.socket.emit('join_room', { room });
    }
  }

  leaveRoom(room) {
    if (this.socket) {
      this.socket.emit('leave_room', { room });
    }
  }
}

export default new WebSocketService();
```
