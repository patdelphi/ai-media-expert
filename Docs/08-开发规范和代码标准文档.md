        self.db = db
        self._cache: Dict[int, Video] = {}
    
    def get_video(self, video_id: int) -> Optional[Video]:
        """获取视频"""
        if video_id in self._cache:

from contextlib import contextmanager

@contextmanager
def database_transaction(db: Session):
    """数据库事务上下文管理器"""
    try:
        yield db
        db.commit()
        logger.info("数据库事务提交成功")
    except Exception as e:
        db.rollback()
        logger.error(f"数据库事务回滚: {e}", exc_info=True)
        raise
    finally:
        db.close()

def setup_logging(log_level: str = "INFO", log_file: str = None) -> None:
    """配置日志系统"""
    
    # 基础配置
    logging.basicConfig(
        level=getattr(logging, log_level.upper()),
        format="%(asctime)s - %(name)s - %(levelname)s - %(message)s",
        handlers=[
            logging.StreamHandler(sys.stdout)
        ]
    )
    
    # 文件日志
    if log_file:
        log_path = Path(log_file)
        log_path.parent.mkdir(parents=True, exist_ok=True)
        
        file_handler = RotatingFileHandler(
            log_file,
            maxBytes=10 * 1024 * 1024,  # 10MB
            backupCount=5
        )
        file_handler.setFormatter(
            logging.Formatter(
                "%(asctime)s - %(name)s - %(levelname)s - %(funcName)s:%(lineno)d - %(message)s"
            )
        )
        logging.getLogger().addHandler(file_handler)
    
    # 配置结构化日志
    structlog.configure(
        processors=[
            structlog.stdlib.filter_by_level,
            structlog.stdlib.add_logger_name,
            structlog.stdlib.add_log_level,
            structlog.stdlib.PositionalArgumentsFormatter(),

                f"耗时: {execution_time:.3f}s, "
                f"错误: {e}"
            )
            raise
    
    return wrapper

import redis
from functools import wraps
import json
import hashlib

# Redis缓存装饰器
def cache_result(expire_time: int = 3600, key_prefix: str = ""):
    """缓存函数结果"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            # 生成缓存键
            cache_key = f"{key_prefix}:{func.__name__}:"
            key_data = f"{args}:{sorted(kwargs.items())}"
            cache_key += hashlib.md5(key_data.encode()).hexdigest()
            
            # 尝试从缓存获取
            cached_result = redis_client.get(cache_key)
            if cached_result:
                return json.loads(cached_result)
            
            # 执行函数并缓存结果
            result = func(*args, **kwargs)
            redis_client.setex(
                cache_key, 
                expire_time, 
                json.dumps(result, default=str)
            )
            
            return result
        return wrapper
    return decorator

# 使用示例
@cache_result(expire_time=1800, key_prefix="video_stats")
def get_video_statistics(video_id: int) -> Dict[str, Any]:
    """获取视频统计信息（带缓存）"""

    return {
        "view_count": video.view_count,
        "like_count": video.like_count,
        "comment_count": video.comment_count,
        "share_count": video.share_count
    }

#### 7.2.1 JWT安全实现
```python
from jose import JWTError, jwt
from datetime import datetime, timedelta
from app.core.config import settings
import secrets

class JWTManager:
    """JWT管理器"""
    
    def __init__(self):
        self.secret_key = settings.SECRET_KEY
        self.algorithm = settings.ALGORITHM
        self.access_token_expire_minutes = settings.ACCESS_TOKEN_EXPIRE_MINUTES
    
    def create_access_token(
        self, 
        data: dict, 
        expires_delta: Optional[timedelta] = None
    ) -> str:
        """创建访问令牌"""
        to_encode = data.copy()
        
        if expires_delta:
            expire = datetime.utcnow() + expires_delta
        else:
            expire = datetime.utcnow() + timedelta(minutes=self.access_token_expire_minutes)
        
        # 添加标准声明
        to_encode.update({
            "exp": expire,
            "iat": datetime.utcnow(),
            "jti": secrets.token_urlsafe(32),  # JWT ID
            "type": "access"
        })
        
        encoded_jwt = jwt.encode(to_encode, self.secret_key, algorithm=self.algorithm)
        return encoded_jwt
    
    def verify_token(self, token: str) -> Optional[dict]:
        """验证令牌"""
        try:
            payload = jwt.decode(
                token, 
                self.secret_key, 
                algorithms=[self.algorithm],
                options={
                    "verify_exp": True,
                    "verify_iat": True,
                    "verify_signature": True
                }
            )
            
            # 检查令牌类型
            if payload.get("type") != "access":
                return None
            
            # 检查是否在黑名单中
            jti = payload.get("jti")
            if jti and self._is_token_blacklisted(jti):
                return None
            
            return payload
            
        except JWTError:
            return None
    
    def blacklist_token(self, token: str) -> bool:
        """将令牌加入黑名单"""
        try:
            payload = jwt.decode(
                token, 
                self.secret_key, 
                algorithms=[self.algorithm],
                options={"verify_exp": False}  # 即使过期也要加入黑名单
            )
            
            jti = payload.get("jti")
            if jti:
                # 存储到Redis，过期时间为令牌的剩余有效期
                exp = payload.get("exp")
                if exp:
                    ttl = max(0, exp - int(datetime.utcnow().timestamp()))
                    redis_client.setex(f"blacklist:{jti}", ttl, "1")
                    return True
            
            return False
            
        except JWTError:
            return False
    
    def _is_token_blacklisted(self, jti: str) -> bool:
        """检查令牌是否在黑名单中"""
        return redis_client.exists(f"blacklist:{jti}")
```

    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            # 从参数中获取当前用户
            current_user = None
            for arg in args:
                if isinstance(arg, User):
                    current_user = arg
                    break
            
            if not current_user:
                raise HTTPException(
                    status_code=status.HTTP_401_UNAUTHORIZED,
                    detail="用户未认证"
                )
            
            # 检查权限
            if not has_permission(current_user, permission):
                raise HTTPException(
                    status_code=status.HTTP_403_FORBIDDEN,
                    detail=f"缺少权限: {permission.value}"
                )
            
            return func(*args, **kwargs)
        return wrapper
    return decorator

def has_permission(user: User, permission: Permission) -> bool:
    """检查用户是否有指定权限"""
    # 管理员拥有所有权限
    if user.role == "admin":
        return True
    
    # 检查角色权限
    role_permissions = get_role_permissions(user.role)
    if permission.value in role_permissions:
        return True
    
    # 检查用户特定权限
    user_permissions = get_user_permissions(user.id)
    return permission.value in user_permissions

# 使用示例
@require_permission(Permission.VIDEO_DELETE)
