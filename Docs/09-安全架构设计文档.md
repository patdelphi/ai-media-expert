## 1. 概述

### 1.1 安全目标
本文档详细描述AI新媒体专家系统的安全架构设计，确保系统在数据保护、访问控制、网络安全、应用安全等方面达到企业级安全标准。

### 1.2 安全原则
- **最小权限原则**：用户和服务只获得完成任务所需的最小权限
- **深度防御**：多层安全防护，避免单点失效
- **零信任架构**：不信任任何内部或外部实体，持续验证
- **数据保护**：全生命周期的数据安全保护
- **合规性**：符合相关法律法规和行业标准

### 1.3 威胁模型
- **外部攻击**：DDoS攻击、SQL注入、XSS攻击、CSRF攻击
- **内部威胁**：权限滥用、数据泄露、恶意操作
- **数据安全**：敏感数据泄露、数据篡改、数据丢失
- **系统安全**：服务中断、系统入侵、恶意软件
- **合规风险**：隐私法规违规、数据跨境传输风险

class MFAService:
    def __init__(self):
        self.totp_service = TOTPService()
        self.sms_service = SMSService()
        self.email_service = EmailService()
    
    def enable_mfa(self, user_id: int, method: str) -> MFASetupResult:
        """启用多因子认证"""
        if method == "totp":
            secret = self.totp_service.generate_secret()
            qr_code = self.totp_service.generate_qr_code(user_id, secret)
            return MFASetupResult(secret=secret, qr_code=qr_code)
        elif method == "sms":
            return self.sms_service.setup_sms_mfa(user_id)
        elif method == "email":
            return self.email_service.setup_email_mfa(user_id)
    
    def verify_mfa(self, user_id: int, code: str, method: str) -> bool:
        """验证多因子认证码"""
        if method == "totp":
            return self.totp_service.verify_code(user_id, code)
        elif method == "sms":
            return self.sms_service.verify_code(user_id, code)
        elif method == "email":
            return self.email_service.verify_code(user_id, code)
        return False

#### 3.1.2 JWT令牌安全
```python
# JWT安全配置
class JWTSecurity:
    def __init__(self):
        self.secret_key = self.get_secret_key()
        self.algorithm = "HS256"
        self.access_token_expire = 30  # 分钟
        self.refresh_token_expire = 7   # 天
        self.blacklist = RedisBlacklist()
    
    def create_access_token(self, user_id: int, permissions: List[str]) -> str:
        """创建访问令牌"""
        payload = {
            "user_id": user_id,
            "permissions": permissions,
            "exp": datetime.utcnow() + timedelta(minutes=self.access_token_expire),
            "iat": datetime.utcnow(),
            "jti": str(uuid.uuid4())  # JWT ID用于黑名单
        }
        return jwt.encode(payload, self.secret_key, algorithm=self.algorithm)
    
    def verify_token(self, token: str) -> TokenPayload:
        """验证令牌"""
        try:
            payload = jwt.decode(token, self.secret_key, algorithms=[self.algorithm])
            
            # 检查黑名单
            if self.blacklist.is_blacklisted(payload.get("jti")):
                raise TokenBlacklistedException("Token is blacklisted")
            
            return TokenPayload(**payload)
        except jwt.ExpiredSignatureError:
            raise TokenExpiredException("Token has expired")
        except jwt.InvalidTokenError:
            raise TokenInvalidException("Invalid token")
    
    def revoke_token(self, token: str) -> bool:
        """撤销令牌"""
        try:
            payload = jwt.decode(token, self.secret_key, algorithms=[self.algorithm])
            jti = payload.get("jti")
            exp = payload.get("exp")
            
            # 添加到黑名单，设置过期时间
            ttl = exp - int(datetime.utcnow().timestamp())
            return self.blacklist.add_to_blacklist(jti, ttl)
        except Exception:
            return False
```

#### 3.2.2 权限检查实现
```python
class PermissionChecker:
    def __init__(self, db: Session):
        self.db = db
        self.cache = RedisCache()
    
    def check_permission(self, user_id: int, resource: str, action: str) -> bool:
        """检查用户权限"""
        cache_key = f"user_permissions:{user_id}"
        permissions = self.cache.get(cache_key)
        
        if not permissions:
            permissions = self._load_user_permissions(user_id)
            self.cache.set(cache_key, permissions, ttl=300)  # 5分钟缓存
        
        required_permission = f"{resource}:{action}"
        return required_permission in permissions
    
    def _load_user_permissions(self, user_id: int) -> Set[str]:
        """加载用户权限"""
        query = """
        SELECT DISTINCT p.resource, p.action
        FROM permissions p
        JOIN role_permissions rp ON p.id = rp.permission_id
        JOIN user_roles ur ON rp.role_id = ur.role_id
        WHERE ur.user_id = %s
        AND (ur.expires_at IS NULL OR ur.expires_at > NOW())
        """
        
        result = self.db.execute(query, (user_id,))
        permissions = set()
        
        for row in result:
            permissions.add(f"{row.resource}:{row.action}")
        
        return permissions

# FastAPI权限装饰器
def require_permission(resource: str, action: str):
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            # 从依赖注入获取当前用户
            current_user = kwargs.get('current_user')
            if not current_user:
                raise HTTPException(status_code=401, detail="Authentication required")
            
            # 检查权限
            permission_checker = PermissionChecker(kwargs.get('db'))
            if not permission_checker.check_permission(current_user.id, resource, action):
                raise HTTPException(status_code=403, detail="Insufficient permissions")
            
            return await func(*args, **kwargs)
        return wrapper
    return decorator
```

# 数据加密服务
class DataEncryption:
    def __init__(self):
        self.key = self._load_encryption_key()
        self.cipher_suite = Fernet(self.key)
    
    def encrypt_sensitive_data(self, data: str) -> str:
        """加密敏感数据"""
        if not data:
            return data
        
        encrypted_data = self.cipher_suite.encrypt(data.encode())
        return base64.b64encode(encrypted_data).decode()
    
    def decrypt_sensitive_data(self, encrypted_data: str) -> str:
        """解密敏感数据"""
        if not encrypted_data:
            return encrypted_data
        
        try:
            decoded_data = base64.b64decode(encrypted_data.encode())
            decrypted_data = self.cipher_suite.decrypt(decoded_data)
            return decrypted_data.decode()
        except Exception as e:
            logger.error(f"Decryption failed: {e}")
            raise DecryptionException("Failed to decrypt data")
    
    def _load_encryption_key(self) -> bytes:
        """加载加密密钥"""
        key_path = os.getenv("ENCRYPTION_KEY_PATH", "/etc/secrets/encryption.key")
        
        if os.path.exists(key_path):
            with open(key_path, 'rb') as key_file:
                return key_file.read()
        else:
            # 生成新密钥（仅用于开发环境）
            key = Fernet.generate_key()
            logger.warning("Generated new encryption key. This should not happen in production.")
            return key

# 数据库字段加密
class EncryptedField:
    def __init__(self, encryption_service: DataEncryption):
        self.encryption_service = encryption_service
    
    def __set_name__(self, owner, name):
        self.name = name
        self.private_name = f'_{name}'
    
    def __get__(self, obj, objtype=None):
        if obj is None:
            return self
        encrypted_value = getattr(obj, self.private_name, None)
        if encrypted_value:
            return self.encryption_service.decrypt_sensitive_data(encrypted_value)
        return None
    
    def __set__(self, obj, value):
        if value:
            encrypted_value = self.encryption_service.encrypt_sensitive_data(value)
            setattr(obj, self.private_name, encrypted_value)
        else:
            setattr(obj, self.private_name, None)

class DataMasking:
    @staticmethod
    def mask_email(email: str) -> str:
        """邮箱脱敏"""
        if not email or '@' not in email:
            return email
        
        local, domain = email.split('@', 1)
        if len(local) <= 2:
            masked_local = '*' * len(local)
        else:
            masked_local = local[0] + '*' * (len(local) - 2) + local[-1]
        
        return f"{masked_local}@{domain}"
    
    @staticmethod
    def mask_phone(phone: str) -> str:
        """手机号脱敏"""
        if not phone or len(phone) < 7:
            return phone
        
        return phone[:3] + '*' * (len(phone) - 6) + phone[-3:]
    
    @staticmethod
    def mask_id_card(id_card: str) -> str:
        """身份证号脱敏"""
        if not id_card or len(id_card) < 8:
            return id_card
        
        return id_card[:4] + '*' * (len(id_card) - 8) + id_card[-4:]
    
    @staticmethod
    def mask_video_content(content: str, max_length: int = 100) -> str:
        """视频内容脱敏"""
        if not content:
            return content
        
        if len(content) <= max_length:
            return content
        
        return content[:max_length] + "...[内容已截断]"

# XSS防护
from markupsafe import escape
import bleach

class XSSProtection:
    # 允许的HTML标签和属性
    ALLOWED_TAGS = ['p', 'br', 'strong', 'em', 'u', 'ol', 'ul', 'li']
    ALLOWED_ATTRIBUTES = {}
    
    @staticmethod
    def sanitize_html(content: str) -> str:
        """清理HTML内容"""
        if not content:
            return content
        
        # 使用bleach清理HTML
        cleaned = bleach.clean(
            content,
            tags=XSSProtection.ALLOWED_TAGS,
            attributes=XSSProtection.ALLOWED_ATTRIBUTES,
            strip=True
        )
        
        return cleaned
    
    @staticmethod
    def escape_user_input(content: str) -> str:
        """转义用户输入"""
        if not content:
            return content
        
        return escape(content)

class APIKeyManager:
    def __init__(self, db: Session):
        self.db = db
    
    def generate_api_key(self, user_id: int, name: str, permissions: List[str]) -> APIKey:

            self.db.commit()

class SecurityMonitor:
    def __init__(self, db: Session, redis_client):
        self.db = db
        self.redis = redis_client
    
    def detect_suspicious_login(self, user_id: int, ip_address: str, user_agent: str) -> bool:
        """检测可疑登录"""
        # 检查IP地理位置变化
        last_ip = self.redis.get(f"user_last_ip:{user_id}")
        if last_ip and last_ip != ip_address:
            if self._is_different_country(last_ip, ip_address):
                self._alert_suspicious_login(user_id, ip_address, "Different country")
                return True
        
        # 检查登录频率
        login_count = self.redis.incr(f"login_attempts:{ip_address}")
        self.redis.expire(f"login_attempts:{ip_address}", 3600)  # 1小时过期
        
        if login_count > 10:  # 1小时内超过10次登录
            self._alert_suspicious_login(user_id, ip_address, "High frequency login")
            return True
        
        # 检查User-Agent变化
        last_ua = self.redis.get(f"user_last_ua:{user_id}")
        if last_ua and last_ua != user_agent:
            # 简单的User-Agent变化检测
            if not self._is_similar_user_agent(last_ua, user_agent):
                self._alert_suspicious_login(user_id, ip_address, "Different user agent")
        
        # 更新用户信息
        self.redis.set(f"user_last_ip:{user_id}", ip_address, ex=86400)
        self.redis.set(f"user_last_ua:{user_id}", user_agent, ex=86400)
        
        return False
    
    def detect_data_exfiltration(self, user_id: int, download_size: int, time_window: int = 3600) -> bool:
        """检测数据泄露"""
        # 统计时间窗口内的下载量
        current_time = int(time.time())
        window_start = current_time - time_window
        
        # 使用Redis有序集合记录下载历史
        key = f"download_history:{user_id}"
        self.redis.zadd(key, {current_time: download_size})
        self.redis.zremrangebyscore(key, 0, window_start)  # 清理过期数据
        
        # 计算时间窗口内总下载量
        total_size = sum([float(score) for score in self.redis.zrange(key, 0, -1, withscores=True)])
        
        # 检查是否超过阈值（例如1GB/小时）
        threshold = 1024 * 1024 * 1024  # 1GB
        if total_size > threshold:
            self._alert_data_exfiltration(user_id, total_size, time_window)
            return True
        
        return False
    
    def _alert_suspicious_login(self, user_id: int, ip_address: str, reason: str):

class GDPRCompliance:
    def __init__(self, db: Session):

        """处理数据主体请求"""

        elif request_type == "deletion":

        elif request_type == "portability":
            return self._export_portable_data(user_id)
        elif request_type == "rectification":

        else:

        for task in tasks:

                "url": task.url,
                "status": task.status,
                "created_at": task.created_at.isoformat()
            })

# 安全事件响应系统
class SecurityIncidentResponse:
    def __init__(self):
        self.severity_levels = {
            "critical": {"response_time": 15, "escalation_time": 30},  # 分钟
            "high": {"response_time": 60, "escalation_time": 120},
            "medium": {"response_time": 240, "escalation_time": 480},
            "low": {"response_time": 1440, "escalation_time": 2880}
        }
    
    def handle_incident(self, incident_data: dict) -> str:
        """处理安全事件"""
        incident_id = self._create_incident(incident_data)
        
        # 根据严重程度自动响应
        severity = incident_data.get("severity", "medium")
        
        if severity == "critical":
            self._critical_incident_response(incident_id, incident_data)
        elif severity == "high":
            self._high_incident_response(incident_id, incident_data)
        else:
            self._standard_incident_response(incident_id, incident_data)
        
        return incident_id
    
    def _critical_incident_response(self, incident_id: str, incident_data: dict):
        """关键事件响应"""
        # 立即通知安全团队
        self._notify_security_team(incident_id, "CRITICAL", incident_data)
        
        # 自动隔离受影响的系统
        if incident_data.get("type") == "malware_detected":
            self._isolate_infected_systems(incident_data.get("affected_hosts", []))
        
        # 启动应急响应流程
        self._activate_emergency_response(incident_id)
