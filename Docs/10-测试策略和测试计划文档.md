## 2. 测试策略

### 2.1 测试金字塔模型

```
                    E2E测试
                   /        \
                  /   UI测试   \
                 /              \
                /________________\
               /                  \
              /    集成测试        \
             /                    \
            /______________________\
           /                        \
          /        单元测试          \
         /                          \
        /____________________________\
```

#### 2.1.1 单元测试（70%）
- **目标**：验证单个函数、方法、类的正确性
- **覆盖率要求**：代码覆盖率 ≥ 85%
- **工具**：pytest, unittest, coverage
- **执行频率**：每次代码提交

#### 2.1.2 集成测试（20%）
- **目标**：验证模块间的接口和交互
- **范围**：API接口、数据库交互、第三方服务集成
- **工具**：pytest, requests, testcontainers
- **执行频率**：每日构建

#### 2.1.3 端到端测试（10%）
- **目标**：验证完整的用户场景和业务流程
- **范围**：关键业务流程、用户界面交互
- **工具**：Playwright, Selenium
- **执行频率**：每周回归测试

#### 2.2.1 环境分层
- **开发环境（DEV）**：开发人员本地测试

```python
import factory
from factory.alchemy import SQLAlchemyModelFactory
from app.models import User, Video, DownloadTask

class UserFactory(SQLAlchemyModelFactory):
    class Meta:
        model = User
        sqlalchemy_session_persistence = "commit"
    
    email = factory.Sequence(lambda n: f"user{n}@example.com")
    username = factory.Sequence(lambda n: f"user{n}")
    full_name = factory.Faker("name")
    password_hash = factory.LazyFunction(lambda: "hashed_password")
    role = "user"
    status = "active"

class VideoFactory(SQLAlchemyModelFactory):
    class Meta:
        model = Video
        sqlalchemy_session_persistence = "commit"
    
    title = factory.Faker("sentence", nb_words=4)
    description = factory.Faker("text", max_nb_chars=200)
    platform = factory.Iterator(["tiktok", "youtube", "bilibili"])
    duration = factory.Faker("random_int", min=30, max=600)
    file_path = factory.Faker("file_path", extension="mp4")
    created_by = factory.SubFactory(UserFactory)

class DownloadTaskFactory(SQLAlchemyModelFactory):
    class Meta:
        model = DownloadTask
        sqlalchemy_session_persistence = "commit"
    
    url = factory.Faker("url")
    status = factory.Iterator(["pending", "processing", "completed", "failed"])
    user = factory.SubFactory(UserFactory)
```

```python
class TestPerformance:
    def test_api_response_time(self, client, auth_headers):
        """测试API响应时间"""
        import time
        
        start_time = time.time()
        response = client.get("/api/v1/videos", headers=auth_headers)
        end_time = time.time()
        
        response_time = end_time - start_time
        
        assert response.status_code == 200
        assert response_time < 0.5  # 响应时间小于500ms
    
    def test_concurrent_requests(self, client, auth_headers):
        """测试并发请求处理"""
        import concurrent.futures
        import threading
        
        def make_request():
            return client.get("/api/v1/videos", headers=auth_headers)
        
        # 并发50个请求
        with concurrent.futures.ThreadPoolExecutor(max_workers=50) as executor:
            futures = [executor.submit(make_request) for _ in range(50)]
            results = [future.result() for future in futures]
        
        # 验证所有请求都成功
        success_count = sum(1 for r in results if r.status_code == 200)
        assert success_count >= 45  # 至少90%成功率
    
    def test_database_query_performance(self, db):
        """测试数据库查询性能"""
        # 创建大量测试数据
        users = [UserFactory() for _ in range(1000)]
        videos = [VideoFactory(created_by=users[i % len(users)]) for i in range(10000)]
        
        db.add_all(users + videos)
        db.commit()
        
        import time

        # 执行复杂查询

            Video.platform == "tiktok",
```

```python
class TestSecurity:
    def test_sql_injection_protection(self, client, auth_headers):

        # 尝试SQL注入攻击
        malicious_input = "'; DROP TABLE users; --"
        
        response = client.get(
            f"/api/v1/videos?search={malicious_input}",
            headers=auth_headers
        )
        
        # 应该正常返回，不会执行恶意SQL
        assert response.status_code in [200, 400]
        # 验证数据库表仍然存在
        assert client.get("/api/v1/users/me", headers=auth_headers).status_code == 200
    
    def test_xss_protection(self, client, auth_headers):
        """测试XSS防护"""
        malicious_script = "<script>alert('XSS')</script>"
        
        video_data = {
            "title": malicious_script,
            "description": "Test video"
        }
        
        response = client.post(
            "/api/v1/videos",
            json=video_data,
            headers=auth_headers
        )
        
        if response.status_code == 201:
            # 获取创建的视频
            video_id = response.json()["id"]

                f"/api/v1/videos/{video_id}",

            # 验证脚本被转义或过滤

            "/api/v1/videos",
```

```python
# 性能监控装饰器
import time
import functools
from prometheus_client import Counter, Histogram, Gauge

# Prometheus指标
REQUEST_COUNT = Counter('http_requests_total', 'Total HTTP requests', ['method', 'endpoint', 'status'])
REQUEST_DURATION = Histogram('http_request_duration_seconds', 'HTTP request duration', ['method', 'endpoint'])
ACTIVE_CONNECTIONS = Gauge('active_connections', 'Active connections')

def monitor_performance(func):
    """性能监控装饰器"""
    @functools.wraps(func)
    async def wrapper(*args, **kwargs):
        start_time = time.time()
        
        try:
            result = await func(*args, **kwargs)
            status = "success"
            return result
        except Exception as e:
            status = "error"
            raise
        finally:
            duration = time.time() - start_time
            
            # 记录指标
            REQUEST_COUNT.labels(
                method=kwargs.get('method', 'unknown'),
                endpoint=kwargs.get('endpoint', 'unknown'),
                status=status
            ).inc()
            
            REQUEST_DURATION.labels(
                method=kwargs.get('method', 'unknown'),
                endpoint=kwargs.get('endpoint', 'unknown')
            ).observe(duration)
    
    return wrapper
```

## 5. 测试报告和质量门禁

### 5.1 测试报告模板

```python
# 测试报告生成器
import json
from datetime import datetime
from typing import Dict, List

class TestReportGenerator:
    def __init__(self):
        self.report_data = {
            "test_run_info": {},
            "test_results": {},
            "coverage_info": {},
            "performance_metrics": {},
            "quality_gates": {}
        }
    
    def generate_report(self, test_results: Dict) -> str:
        """生成测试报告"""
        self.report_data["test_run_info"] = {
            "timestamp": datetime.now().isoformat(),
            "environment": test_results.get("environment", "test"),
            "version": test_results.get("version", "unknown"),
            "duration": test_results.get("duration", 0)
        }
        
        self.report_data["test_results"] = {
            "total_tests": test_results.get("total", 0),
            "passed": test_results.get("passed", 0),
            "failed": test_results.get("failed", 0),
            "skipped": test_results.get("skipped", 0),
            "pass_rate": self._calculate_pass_rate(test_results)
        }
        
        self.report_data["coverage_info"] = test_results.get("coverage", {})
        self.report_data["performance_metrics"] = test_results.get("performance", {})
        
        # 质量门禁检查
        self.report_data["quality_gates"] = self._check_quality_gates()
        
        return self._format_report()
    
    def _calculate_pass_rate(self, test_results: Dict) -> float:
        """计算通过率"""
        total = test_results.get("total", 0)
        passed = test_results.get("passed", 0)
        
        if total == 0:
            return 0.0
        
        return (passed / total) * 100
    
    def _check_quality_gates(self) -> Dict:
        """检查质量门禁"""
        gates = {
            "test_pass_rate": {
                "threshold": 95.0,
                "actual": self.report_data["test_results"].get("pass_rate", 0),
                "status": "unknown"
            },
            "code_coverage": {
                "threshold": 85.0,
                "actual": self.report_data["coverage_info"].get("total", 0),
                "status": "unknown"
            },
            "performance": {
                "threshold": 500,  # ms
                "actual": self.report_data["performance_metrics"].get("avg_response_time", 0),
                "status": "unknown"
            }
        }
        
        # 检查每个门禁
        for gate_name, gate_info in gates.items():
            if gate_name == "performance":
                # 性能指标：实际值应小于阈值
                gate_info["status"] = "pass" if gate_info["actual"] <= gate_info["threshold"] else "fail"
            else:
                # 其他指标：实际值应大于等于阈值
                gate_info["status"] = "pass" if gate_info["actual"] >= gate_info["threshold"] else "fail"
        
        return gates
    
    def _format_report(self) -> str:
        """格式化报告"""
        report = f"""
# 测试报告

## 测试运行信息
- 时间: {self.report_data['test_run_info']['timestamp']}
- 环境: {self.report_data['test_run_info']['environment']}
- 版本: {self.report_data['test_run_info']['version']}
- 持续时间: {self.report_data['test_run_info']['duration']}秒

## 测试结果
- 总测试数: {self.report_data['test_results']['total_tests']}
- 通过: {self.report_data['test_results']['passed']}
- 失败: {self.report_data['test_results']['failed']}
- 跳过: {self.report_data['test_results']['skipped']}
- 通过率: {self.report_data['test_results']['pass_rate']:.2f}%

## 代码覆盖率
- 总覆盖率: {self.report_data['coverage_info'].get('total', 0):.2f}%
- 语句覆盖率: {self.report_data['coverage_info'].get('statements', 0):.2f}%
- 分支覆盖率: {self.report_data['coverage_info'].get('branches', 0):.2f}%

## 质量门禁
"""
        
        for gate_name, gate_info in self.report_data["quality_gates"].items():
            status_emoji = "✅" if gate_info["status"] == "pass" else "❌"
            report += f"- {gate_name}: {status_emoji} {gate_info['actual']} (阈值: {gate_info['threshold']})\n"
        
        return report
```
