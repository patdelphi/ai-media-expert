## 1. 概述

### 1.1 配置管理目标
本文档规范AI新媒体专家系统的配置管理策略，确保配置的一致性、安全性、可维护性和可追溯性，支持多环境部署和动态配置更新。

### 1.2 配置管理原则
- **环境隔离**：不同环境使用独立的配置
- **安全优先**：敏感配置加密存储和传输
- **版本控制**：配置变更可追溯和回滚
- **自动化**：配置部署和更新自动化
- **标准化**：统一的配置格式和命名规范

### 1.3 适用范围
- 应用程序配置
- 数据库连接配置
- 第三方服务配置
- 环境变量管理
- 密钥和证书管理

## 2. 配置分类和层次

### 2.1 配置分类

#### 2.1.1 按敏感性分类
- **公开配置**：可以公开的配置项（如端口号、超时时间）
- **内部配置**：内部使用的配置项（如服务地址、缓存配置）
- **敏感配置**：需要保护的配置项（如密码、API密钥、证书）

#### 2.1.2 按变更频率分类
- **静态配置**：很少变更的配置（如数据库结构配置）
- **动态配置**：可能经常变更的配置（如业务参数、开关配置）
- **运行时配置**：运行时可动态调整的配置（如日志级别、缓存大小）

#### 2.1.3 按作用域分类
- **全局配置**：影响整个系统的配置
- **服务配置**：特定服务的配置
- **模块配置**：特定模块的配置
- **用户配置**：用户个性化配置

### 3.2 配置加载机制

#### 3.2.1 配置加载器
```python
# app/core/config_loader.py
import os
import yaml
from pathlib import Path
from typing import Dict, Any, Optional
from pydantic import BaseSettings
from dotenv import load_dotenv

class ConfigLoader:
    def __init__(self, config_dir: str = "config"):
        self.config_dir = Path(config_dir)
        self.environment = os.getenv("ENVIRONMENT", "development")
        
    def load_config(self) -> Dict[str, Any]:
        """加载完整配置"""
        # 1. 加载默认配置
        config = self._load_yaml_config("defaults.yml")
        
        # 2. 加载主配置文件
        app_config = self._load_yaml_config("app.yml")
        config = self._merge_config(config, app_config)
        
        # 3. 加载环境特定配置
        env_config = self._load_yaml_config(f"environments/{self.environment}.yml")
        config = self._merge_config(config, env_config)
        
        # 4. 加载环境变量
        self._load_env_variables()
        
        # 5. 应用环境变量覆盖
        config = self._apply_env_overrides(config)
        
        return config
    
    def _load_yaml_config(self, filename: str) -> Dict[str, Any]:
        """加载YAML配置文件"""
        file_path = self.config_dir / filename
        
        if not file_path.exists():
            return {}
        
        with open(file_path, 'r', encoding='utf-8') as f:
            return yaml.safe_load(f) or {}
    
    def _load_env_variables(self):
        """加载环境变量文件"""
        # 加载通用环境变量
        env_file = Path(".env")
        if env_file.exists():
            load_dotenv(env_file)
        
        # 加载环境特定的环境变量
        env_specific_file = Path(f".env.{self.environment}")
        if env_specific_file.exists():
            load_dotenv(env_specific_file, override=True)
    
    def _merge_config(self, base: Dict[str, Any], override: Dict[str, Any]) -> Dict[str, Any]:
        """合并配置字典"""
        result = base.copy()
        
        for key, value in override.items():
            if key in result and isinstance(result[key], dict) and isinstance(value, dict):
                result[key] = self._merge_config(result[key], value)
            else:
                result[key] = value
        
        return result
    
    def _apply_env_overrides(self, config: Dict[str, Any]) -> Dict[str, Any]:
        """应用环境变量覆盖"""
        # 定义环境变量到配置路径的映射
        env_mappings = {
            "DATABASE_URL": "app.database.url",
            "REDIS_URL": "app.redis.url",
            "JWT_SECRET_KEY": "app.security.jwt_secret_key",
            "LOG_LEVEL": "app.logging.level",
            "DEBUG": "app.server.debug",
            "WORKERS": "app.server.workers"
        }
        
        for env_var, config_path in env_mappings.items():
            env_value = os.getenv(env_var)
            if env_value is not None:
                self._set_nested_value(config, config_path, env_value)
        
        return config
    
    def _set_nested_value(self, config: Dict[str, Any], path: str, value: Any):
        """设置嵌套配置值"""
        keys = path.split('.')
        current = config
        
        for key in keys[:-1]:
            if key not in current:
                current[key] = {}
            current = current[key]
        
        # 类型转换
        if isinstance(value, str):
            if value.lower() in ('true', 'false'):
                value = value.lower() == 'true'
            elif value.isdigit():
                value = int(value)
            elif value.replace('.', '').isdigit():
                value = float(value)
        
        current[keys[-1]] = value
```

        # 嵌套模型的环境变量

# 配置验证器
class ConfigValidator:
    def __init__(self, config_data: Dict[str, Any]):
        self.config_data = config_data
    
    def validate(self) -> AppConfig:
        """验证配置"""
        try:
            # 展平嵌套配置以适应Pydantic
            flattened_config = self._flatten_config(self.config_data)
            
            # 创建并验证配置对象
            app_config = AppConfig(**flattened_config)
            
            # 额外的业务逻辑验证
            self._validate_business_rules(app_config)
            
            return app_config
            
        except Exception as e:
            raise ValueError(f"Configuration validation failed: {e}")
    
    def _flatten_config(self, config: Dict[str, Any], prefix: str = "") -> Dict[str, Any]:
        """展平嵌套配置"""
        flattened = {}
        
        for key, value in config.items():
            new_key = f"{prefix}__{key}" if prefix else key
            
            if isinstance(value, dict):
                flattened.update(self._flatten_config(value, new_key))
            else:
                flattened[new_key] = value
        
        return flattened
    
    def _validate_business_rules(self, config: AppConfig):
        """验证业务规则"""
        # 生产环境必须关闭调试模式
        if config.environment == Environment.PRODUCTION and config.debug:

        if config.environment == Environment.PRODUCTION and not config.security.secure_cookies:
            raise ValueError("Secure cookies must be enabled in production")
        
        # 验证AI模型配置
        if config.ai_models.use_local_models:
            import os
            if not os.path.exists(config.ai_models.yolo_model_path):
                raise ValueError(f"YOLO model file not found: {config.ai_models.yolo_model_path}")

#### 4.2.1 配置文件加密

# 配置加密工具
from cryptography.fernet import Fernet
import base64
import json

class ConfigEncryption:
    def __init__(self, key: bytes = None):
        self.key = key or Fernet.generate_key()
        self.cipher_suite = Fernet(self.key)
    
    def encrypt_config(self, config_data: Dict[str, Any]) -> str:
        """加密配置数据"""
        json_data = json.dumps(config_data)
        encrypted_data = self.cipher_suite.encrypt(json_data.encode())
        return base64.b64encode(encrypted_data).decode()
    
    def decrypt_config(self, encrypted_config: str) -> Dict[str, Any]:
        """解密配置数据"""
        encrypted_data = base64.b64decode(encrypted_config.encode())
        decrypted_data = self.cipher_suite.decrypt(encrypted_data)
        return json.loads(decrypted_data.decode())
    
    def encrypt_file(self, input_file: str, output_file: str):
        """加密配置文件"""
        with open(input_file, 'r') as f:
            config_data = yaml.safe_load(f)
        
        encrypted_config = self.encrypt_config(config_data)
        
        with open(output_file, 'w') as f:
            f.write(encrypted_config)
    
    def decrypt_file(self, encrypted_file: str, output_file: str):
        """解密配置文件"""
        with open(encrypted_file, 'r') as f:
            encrypted_config = f.read()
        
        config_data = self.decrypt_config(encrypted_config)
        
        with open(output_file, 'w') as f:
            yaml.dump(config_data, f, default_flow_style=False)
```

# 配置中心客户端
import asyncio
import aiohttp
from typing import Dict, Any, Callable, Optional
from datetime import datetime, timedelta

class ConfigCenter:
    def __init__(self, server_url: str, app_name: str, environment: str):
        self.server_url = server_url
        self.app_name = app_name
        self.environment = environment
        self.config_cache = {}
        self.last_update = None
        self.update_callbacks = []
    
    async def get_config(self, key: str, default: Any = None) -> Any:
        """获取配置项"""
        if key in self.config_cache:
            return self.config_cache[key]
        
        # 从配置中心获取
        config_value = await self._fetch_config(key)
        if config_value is not None:
            self.config_cache[key] = config_value
            return config_value
        
        return default
    
    async def set_config(self, key: str, value: Any) -> bool:
        """设置配置项"""
        try:
            success = await self._update_config(key, value)
            if success:
                self.config_cache[key] = value
                await self._notify_callbacks(key, value)
            return success
        except Exception as e:
            logger.error(f"Failed to set config {key}: {e}")
            return False
    
    async def watch_config(self, key: str, callback: Callable[[str, Any], None]):
        """监听配置变更"""
        self.update_callbacks.append((key, callback))
    
    async def start_polling(self, interval: int = 30):
        """启动配置轮询"""
        while True:
            try:
                await self._poll_updates()
                await asyncio.sleep(interval)
            except Exception as e:
                logger.error(f"Config polling error: {e}")
                await asyncio.sleep(interval)
    
    async def _fetch_config(self, key: str) -> Optional[Any]:
        """从配置中心获取配置"""
        url = f"{self.server_url}/api/v1/config/{self.app_name}/{self.environment}/{key}"
        
        async with aiohttp.ClientSession() as session:
            async with session.get(url) as response:
                if response.status == 200:
                    data = await response.json()
                    return data.get('value')
                return None
    
    async def _update_config(self, key: str, value: Any) -> bool:
        """更新配置到配置中心"""
        url = f"{self.server_url}/api/v1/config/{self.app_name}/{self.environment}/{key}"
        
        async with aiohttp.ClientSession() as session:
            async with session.put(url, json={'value': value}) as response:
                return response.status == 200
    
    async def _poll_updates(self):
        """轮询配置更新"""
        url = f"{self.server_url}/api/v1/config/{self.app_name}/{self.environment}/updates"
        
        params = {}
        if self.last_update:
            params['since'] = self.last_update.isoformat()
        
        async with aiohttp.ClientSession() as session:
            async with session.get(url, params=params) as response:
                if response.status == 200:
                    updates = await response.json()
                    
                    for update in updates.get('changes', []):
                        key = update['key']
                        value = update['value']
                        
                        # 更新缓存
                        self.config_cache[key] = value
                        
                        # 通知回调
                        await self._notify_callbacks(key, value)
                    
                    self.last_update = datetime.now()
    
    async def _notify_callbacks(self, key: str, value: Any):
        """通知配置变更回调"""
        for callback_key, callback in self.update_callbacks:
            if callback_key == key or callback_key == '*':
                try:
                    await callback(key, value)
                except Exception as e:
                    logger.error(f"Config callback error: {e}")
```

# 配置热更新管理器
class ConfigHotReload:
    def __init__(self, app_config: AppConfig):
        self.app_config = app_config
        self.reload_handlers = {}
    
    def register_handler(self, config_path: str, handler: Callable[[Any], None]):
        """注册配置变更处理器"""
        self.reload_handlers[config_path] = handler
    
    async def handle_config_change(self, key: str, new_value: Any):
        """处理配置变更"""
        # 更新配置对象
        self._update_config_value(key, new_value)
        
        # 调用相应的处理器
        if key in self.reload_handlers:
            try:
                await self.reload_handlers[key](new_value)
                logger.info(f"Config {key} updated successfully")
            except Exception as e:
                logger.error(f"Failed to handle config change for {key}: {e}")
    
    def _update_config_value(self, key: str, value: Any):
        """更新配置值"""
        keys = key.split('.')
        current = self.app_config
        
        for k in keys[:-1]:
            current = getattr(current, k)
        
        setattr(current, keys[-1], value)

# 使用示例

    echo "Restarting production services..."
    docker-compose -f docker-compose.prod.yml restart app
else
    echo "Restarting $ENVIRONMENT services..."
    docker-compose restart
fi

# 5. 验证部署
echo "Verifying deployment..."
sleep 10
curl -f http://localhost:8000/health || {
    echo "Health check failed! Rolling back..."
    cp config/environments/$ENVIRONMENT.yml.backup.* config/environments/$ENVIRONMENT.yml
    docker-compose restart
    exit 1
}

from jinja2 import Environment, FileSystemLoader
import yaml

class ConfigTemplateRenderer:
    def __init__(self, template_dir: str = "config/templates"):
        self.env = Environment(loader=FileSystemLoader(template_dir))
    
    def render_config(self, template_name: str, variables: Dict[str, Any]) -> str:
        """渲染配置模板"""
        template = self.env.get_template(template_name)
        return template.render(**variables)
    
    def render_to_file(self, template_name: str, output_file: str, variables: Dict[str, Any]):
        """渲染模板到文件"""
        rendered_content = self.render_config(template_name, variables)
        
        with open(output_file, 'w', encoding='utf-8') as f:
            f.write(rendered_content)
    
    def validate_rendered_config(self, rendered_content: str) -> bool:
        """验证渲染后的配置"""
        try:
            yaml.safe_load(rendered_content)
            return True
        except yaml.YAMLError as e:
            logger.error(f"Invalid YAML in rendered config: {e}")
            return False

class ConfigHealthChecker:

    async def run_health_checks(self) -> Dict[str, Any]:
        """运行所有健康检查"""
        results = {
            "timestamp": datetime.utcnow().isoformat(),
            "overall_status": "healthy",
            "checks": {}
        }
        
        for name, check_func in self.health_checks:
            try:
                start_time = time.time()
                is_healthy = await check_func()
                duration = time.time() - start_time
                
                results["checks"][name] = {
                    "status": "healthy" if is_healthy else "unhealthy",
                    "duration_ms": round(duration * 1000, 2)
                }
                
                if not is_healthy:
                    results["overall_status"] = "unhealthy"
                    
            except Exception as e:
                results["checks"][name] = {
                    "status": "error",
                    "error": str(e)
                }
                results["overall_status"] = "unhealthy"
