### 2. 配置验证
```typescript
const validateDownloadConfig = (config: Partial<DownloadSystemConfig>): ValidationResult => {
  const errors: string[] = [];
  
  // 验证存储路径
  if (config.storage_config?.basePath && !isValidPath(config.storage_config.basePath)) {
    errors.push('存储路径无效');
  }
  
  // 验证并发数量
  if (config.download_settings?.maxConcurrentDownloads) {
    const max = config.download_settings.maxConcurrentDownloads;
    if (max < 1 || max > 10) {
      errors.push('并发下载数量必须在1-10之间');
    }
  }
  
  // 验证文件大小限制
  if (config.download_settings?.maxFileSize) {
    const size = config.download_settings.maxFileSize;
    if (size < 1 || size > 10240) { // 10GB
      errors.push('文件大小限制必须在1MB-10GB之间');
    }
  }
  
  return {
    valid: errors.length === 0,
    errors
  };
};
```

### 3. 配置持久化
```typescript
const saveDownloadConfig = async (config: DownloadSystemConfig): Promise<void> => {
  try {
    // 验证配置
    const validation = validateDownloadConfig(config);
    if (!validation.valid) {
      throw new Error(`配置验证失败: ${validation.errors.join(', ')}`);
    }
    
    // 保存到后端
    await systemConfigService.updateConfigs(config);
    
    // 更新本地状态
    setConfigs(prev => ({ ...prev, ...config }));
    
    toast.success('下载配置保存成功');
  } catch (error) {
    console.error('保存下载配置失败:', error);
    toast.error('保存配置失败');
  }
};
```

interface ConfigVersion {

const migrateDownloadConfig = (oldConfig: any, targetVersion: string): DownloadSystemConfig => {
  // 实现配置迁移逻辑
  switch (targetVersion) {
    case '1.1.0':
      return migrateToV1_1_0(oldConfig);
    case '1.2.0':
      return migrateToV1_2_0(oldConfig);
    default:
      return oldConfig;
  }
};
